import java.awt.*;
import java.awt.image.BufferedImage;
import java.io.File;
import java.io.IOException;
import javax.swing.*;
import javax.imageio.ImageIO;
//Importing all the necessary Packages
public class Initialize extends JFrame {
	JFrame frame = new JFrame("Aliengame");
	static JButton[][] buttons = new JButton[8][8];
	//Creating the JButton field
	
	public static void main(String[] args) throws IOException, InterruptedException {
		new Initialize();
		ButtonMethods.main(args);
	}
	
	public Initialize() throws IOException {
		framesetupstart(); 		//Sets up basic configuration for the window(resizable...)
		paintbackground(); 		//Draws the background to the scene
		frame.setLayout(null);	//Sets layout to null to add JButtons
		buttonssetup();			//Sets up the JButtons (position and size)
		framesetupend();		//Sets up size and Location of the window
	}
	
	public static JButton retbutton(int x, int y) {
		return buttons[x][y];
	}
	//A function to return the button array in order to use the button array in other Classes
	void framesetupstart() {
		frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
		frame.setVisible(true);
		frame.setResizable(false);
	}
	void paintbackground() throws IOException {
		frame.setContentPane(new JPanel() {
			BufferedImage image = ImageIO.read(new File("C:/Users/maxan/Desktop/Eclipse Workspace/Map.jpg"));
			public void paintComponent(Graphics g) {
				super.paintComponent(g);
				g.drawImage(image,0,0,695,695,this);
			}
		});
	}
	void buttonssetup() {
		int x = 10;
		int y = 10;
		for(int i=0;i<8;i++) {
			for(int j=0;j<8;j++) {
				buttons[i][j] = new JButton("Bnr."+ i + j);
				buttons[i][j].setLocation(x, y);
				buttons[i][j].setSize(80,80);
				x = x + 85;
			}
			y = y + 85;
			x = 10;
		}
		for(int a=0;a<8;a++) {
			for(int b=0;b<8;b++) {
				frame.add(buttons[a][b]);
			}
		}
	}
	void framesetupend() {
		frame.setSize(710,730);
		frame.setLocationRelativeTo(null);
		frame.setLayout(null);
	}
}
